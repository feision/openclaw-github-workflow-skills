## 案例1：AI 助手自动创建项目模板

### 场景
AI 助手需要为新项目快速创建基础模板

### 解决方案
```python
import urllib.request, json, os, base64

def create_project_template(owner, repo_name):
    # 创建仓库
    repo_data = json.dumps({
        "name": repo_name,
        "description": "Project template",
        "private": False,
        "auto_init": True
    }).encode()
    
    req = urllib.request.Request('https://gateway.maton.ai/github/user/repos', data=repo_data, method='POST')
    req.add_header('Authorization', f'Bearer {os.environ["MATON_API_KEY"]}')
    req.add_header('Content-Type', 'application/json')
    
    repo = json.load(urllib.request.urlopen(req))
    
    # 上传模板文件
    template_files = {
        "README.md": "# Project Template\n\nGenerated by AI assistant",
        ".gitignore": "*.log\n*.tmp\nnode_modules/\n.env\n",
        "LICENSE": "MIT License",
        "CONTRIBUTING.md": "# Contributing Guidelines\n\nWelcome contributors!"
    }
    
    for filename, content in template_files.items():
        encoded_content = base64.b64encode(content.encode()).decode()
        upload_data = json.dumps({
            "message": f"Add {filename}",
2            "content": encoded_content,
            "branch": "main"
        }).encode()
        
        req = urllib.request.Request(f'https://gateway.maton.ai/github/repos/{repo["owner"]["login"]}/{repo_name}/contents/{filename}', 
                                    data=upload_data, method='PUT')
        req.add_header('Authorization', f'Bearer {os.environ["MATON_API_KEY"]}')
        req.add_header('Content-Type', 'application/json')
        
        try:
            urllib.request.urlopen(req)
            print(f"Uploaded {filename}")
        except Exception as e:
            print(f"Failed to upload {filename}: {e}")
    
    return repo["full_name"]
```

## 案例5：跨仓库自动化同步

### 场景
AI 助手保持多个仓库间的文件同步

### 解决方案
```python
def sync_repositories(source_owner, source_repo, dest_owner, dest_repo):
    # 获取源仓库文件列表
    req = urllib.request.Request(f'https://gateway.maton.ai/github/repos/{source_owner}/{source_repo}/contents')
    req.add_header('Authorization', f'Bearer {os.environ["MATON_API_KEY"]}')
    source_files = json.load(urllib.request.urlopen(req))
    
    for file_info in source_files:
        if file_info["type"] == "file":
            file_path = file_info["path"]
            
            # 获取源文件内容
            req = urllib.request.Request(f'https://gateway.maton.ai/github/repos/{source_owner}/{source_repo}/contents/{file_path}')
            req.add_header('Authorization', f'Bearer {os.environ["MATON_API_KEY"]}')
            source_content = json.load(urllib.request.urlopen(req))
            
            # 上传到目标仓库
            upload_data = json.dumps({
                "message": f"Sync from {source_repo}: {file_path}",
                "content": source_content["content"],
                "branch": "main"
            }).encode()
            
            req = urllib.request.Request(f'https://gateway.maton.ai/github/repos/{dest_owner}/{dest_repo}/contents/{file_path}', 
                                        data=upload_data, method='PUT')
            req.add_header('Authorization', f'Bearer {os.environ["MATON_API_KEY"]}')
            req.add_header('Content-Type', 'application/json')
            
            try:
                urllib.request.urlopen(req)
                print(f"Synced {file_path}")
            except Exception as e:
                print(f"Failed to sync {file_path}: {e}")
```
